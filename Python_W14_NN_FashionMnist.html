PyTorch Lightning Model¶

To define the PyTorch Lightning model, we need to implement a class with the following functions:

  • __init__: Set the model's attributes and parameters
  • forward: How to obtain predictions
  • training_step: Compute the loss for each batch of observations when training
  • test_step: Compute the loss for each batch of observations when testing
  • configure_optimizers: Specify which optimizers to use
In [9]:
import torch
from torch import nn, optim
import lightning.pytorch as pl

class FeedForward(pl.LightningModule):
  def __init__(self, input_dim = 28*28, n_classes = 10, 
               hidden_dim = 128, learning_rate = 5e-4):
    super().__init__()
    self.hidden_dim = hidden_dim
    self.input_dim = input_dim
    self.n_classes = n_classes
    self.layer = nn.Linear(input_dim, hidden_dim)
    self.output = nn.Linear(hidden_dim, n_classes)

    self.learning_rate = learning_rate

  def forward(self, x):
    x = nn.Flatten()(x)
    x = self.layer(x)
    x = nn.ReLU()(x)
    x = self.output(x)
    return x

  def training_step(self, batch, batch_idx):
    x, y = batch
    logits = self.forward(x)
    loss = nn.CrossEntropyLoss()(logits, y)
    self.log("train_loss", loss, prog_bar = True, on_epoch = True, on_step = False)
    return loss

  def test_step(self, batch, batch_idx):
    x, y = batch
    logits = self.forward(x)
    loss = nn.CrossEntropyLoss()(logits, y)
    acc = (torch.argmax(logits, dim=1).float() == y).float().mean()
    self.log("test_loss", loss, on_epoch = True, on_step = False)
    self.log("test_acc", acc, prog_bar = True, on_epoch = True, on_step = False)
    return loss

  def configure_optimizers(self):
    optimizer = optim.AdamW(self.parameters(), lr = self.learning_rate)
    return optimizer

from lightning.pytorch import Trainer
import warnings

csv_logger = pl_loggers.CSVLogger(name = name, save_dir="")
trainer = Trainer(logger=csv_logger, max_epochs = 40)
model = FeedForward()
trainer.fit(model = model, train_dataloaders = train_loader)
