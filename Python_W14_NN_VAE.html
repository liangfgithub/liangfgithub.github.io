Define Network¶






















First, we implement the VAE loss function.


















In [1]:






from torch import nn
import torch

def VAE_loss(x, x_hat, z_mean, z_logvar):
    x = nn.Flatten()(x)
    x_hat = nn.Flatten()(x_hat)
    xent_loss = nn.BCELoss(reduction = 'none')(x_hat, x).mean(1)
    kl_loss = kl_loss = -5e-4 * (1+ z_logvar - 
                                 (z_mean**2) - z_logvar.exp()).mean(1)
    return (xent_loss + kl_loss).mean()

























To implement the network, we use the torchshape package to compute the shape of the input after being passed through the convolutional layers.


















In [6]:






from torchshape import tensorshape
def compute_shape(funcs, shape):
  for f in funcs:
    shape = tensorshape(f, shape)
  return shape

























We map each $28 \times 28$ image to a two-dimensional Gaussian distribution  with a two dimensional mean (z_mean) and two dimensional variance (z_logvar.exp()).

from torch import optim
import lightning.pytorch as pl

class Variational(pl.LightningModule):
    def __init__(self, input_shape = (1, 1, 28, 28), out_channels1 = 32,
                 out_channels2 = 64, kernel_size = (3,3), stride = (2,2), 
                 encode_hidden_dim = 32, latent_dim = 2, learning_rate = 1e-3):
        super(Variational, self).__init__()
        self.learning_rate = learning_rate
        self.input_shape = input_shape
        self.latent_dim = latent_dim
        self.input_channels = input_shape[1]
        
        #Encoding
        self.encode_layer1 = nn.Conv2d(self.input_channels, out_channels1, 
                                       kernel_size, padding = 'same')
        self.encode_layer2 = nn.Conv2d(out_channels1, out_channels2,
                                       kernel_size, stride = stride)
        self.encode_layer3 = nn.Conv2d(out_channels2, out_channels2,
                                       kernel_size, padding = 'same')
        self.encode_layer4 = nn.Conv2d(out_channels2, out_channels2, 
                                       kernel_size, padding = 'same')

        self.shape_before_flatten = compute_shape(
            [self.encode_layer1, self.encode_layer2,
             self.encode_layer3, self.encode_layer4], input_shape)
        self.shape_after_flatten = tensorshape(nn.Flatten(),
                                               self.shape_before_flatten)

        self.encode_layer5 = nn.Linear(self.shape_after_flatten[1],
                                       encode_hidden_dim)

        self.mean_layer = nn.Linear(encode_hidden_dim, latent_dim)
        self.logvar_layer = nn.Linear(encode_hidden_dim, latent_dim)

        #Decoding
        self.decode_layer1 = nn.Linear(latent_dim, self.shape_after_flatten[1])
        self.decode_layer2 = nn.ConvTranspose2d(out_channels2, out_channels1,
                                                kernel_size, stride = stride,
                                                output_padding = 1)
        self.decode_layer3 = nn.Conv2d(out_channels1, self.input_channels,
                                       kernel_size, padding = 'same')

    def encode(self, x):
        x = self.encode_layer1(x)
        x = nn.ReLU()(x)
        x = self.encode_layer2(x)
        x = nn.ReLU()(x)
        x = self.encode_layer3(x)
        x = nn.ReLU()(x)
        x = self.encode_layer4(x)
        x = nn.ReLU()(x)
        x = nn.Flatten()(x)
        x = self.encode_layer5(x)
        x = nn.ReLU()(x)

        z_mean = self.mean_layer(x)
        z_logvar = self.logvar_layer(x)
        return z_mean, z_logvar

    def decode_sample(self, z):
        z = self.decode_layer1(z)
        z = z.reshape((-1, self.shape_before_flatten[1], 
                       self.shape_before_flatten[2], 
                       self.shape_before_flatten[3]))
        z = self.decode_layer2(z)
        z = nn.ReLU()(z)
        z = self.decode_layer3(z)
        x_hat = nn.Sigmoid()(z)
        return x_hat

    def decode(self, z_mean, z_logvar):
        eps = torch.rand(self.latent_dim, 
                         device = 'cuda' if torch.cuda.is_available() else 'cpu')
        z = z_mean + torch.exp(z_logvar)*eps
        x_hat = self.decode_sample(z)
        return x_hat

    def forward(self, x):
        z_mean, z_logvar = self.encode(x)
        x_hat = self.decode(z_mean, z_logvar)
        return x_hat, z_mean, z_logvar

    def training_step(self, batch, batch_idx):
      x, y = batch
      x_hat, z_mean, z_logvar = self.forward(x)
      loss = VAE_loss(x, x_hat, z_mean, z_logvar)
      self.log("train_loss", loss, prog_bar = True, 
               on_step = False, on_epoch = True)
      return loss

    def configure_optimizers(self):
      optimizer = optim.Adam(self.parameters(), lr = self.learning_rate)
      return optimizer
